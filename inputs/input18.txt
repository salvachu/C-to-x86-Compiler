#include <stdio.h>

long accumulate(long base) {
    long sum = 0;
    for (long i = 1; i <= base; i += 1) {
        sum += i * i;
        if (sum > 1000000) {
            sum -= 500000;
        }
    }
    return sum;
}

int main() {
    long start = 100;
    long accumulated = accumulate(start);
    long final_result = accumulated * 2 + start;
    
    printf("%ld", start);
    printf("%ld", accumulated);
    printf("%ld", final_result);
    
    return 0;
}
