#include <stdio.h>

int main() {
    int i = 0;
    unsigned sum = 0;
    long product = 1;
    
    while (i < 5) {
        sum += i * 2 + 3;
        product += sum;
        i += 1;
    }
    
    printf("%d", i);
    printf("%u", sum);
    printf("%ld", product);
    
    return 0;
}
