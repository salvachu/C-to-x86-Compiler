#include <stdio.h>

int main() {
    int x;
    int y;
    unsigned z;
    long w;
    
    x = 5;
    y = 10;
    z = 100;
    w = 500;
    
    x = x + y;
    y = y - 3;
    z += x;
    w -= y;
    
    x += z;
    y -= w;
    
    printf("%d", x);
    printf("%d", y);
    printf("%u", z);
    printf("%ld", w);
    
    return 0;
}
