#include <stdio.h>

int calculate(int base, unsigned multiplier) {
    int result = base * multiplier;
    result += base * 2;
    result -= multiplier;
    return result;
}

int main() {
    int x = 5;
    unsigned y = 3;
    int total = calculate(x, y);
    
    printf("%d", x);
    printf("%u", y);
    printf("%d", total);
    
    return 0;
}
